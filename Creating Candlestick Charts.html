#Function to save figure
import time,os
def figsave(horizan):
    if os.path.exists('{}_{}.png'.format(save,horizan)):
        fname='{}_{}_{}.png'.format(save,horizan,int(time.time()))
    else:
        fname='{}_{}.png'.format(save,horizan)
    return fname
